001package net.peierls.expr;
002
003import static net.peierls.expr.OperatorBuilder.*;
004
005
006/**
007 * Trigonometric operations on expressions.
008 */
009@SuppressWarnings("ImmutableEnumChecker")
010public enum TrigonometricOperator implements DelegatingOperator {
011    /** Sine operator */
012    SINE        (prefix("sin ", Math::sin)      .precedence(100)),
013    /** Cosine operator */
014    COSINE      (prefix("cos ", Math::cos)      .precedence(100)),
015    /** Tangent operator */
016    TANGENT     (prefix("tan ", Math::tan)      .precedence(100)),
017    /** Arcsine operator */
018    ARC_SINE    (prefix("asin ", Math::asin)    .precedence(100)),
019    /** Arccosine operator */
020    ARC_COSINE  (prefix("acos ", Math::acos)    .precedence(100)),
021    /** Arctangent operator */
022    ARC_TANGENT (prefix("atan ", Math::atan)    .precedence(100)),
023    ;
024    final Operator delegate;
025    TrigonometricOperator(Operator delegate) { this.delegate = delegate; }
026    @Override public Operator delegate() { return delegate; }
027
028
029    /**
030     * Returns an expression representing the sine of
031     * the given expression.
032     */
033    public static UnaryOpExpression sin(Expression expr) {
034        return new UnaryOpExpression(SINE, expr);
035    }
036
037    /**
038     * Returns an expression representing the cosine of
039     * the given expression.
040     */
041    public static UnaryOpExpression cos(Expression expr) {
042        return new UnaryOpExpression(COSINE, expr);
043    }
044
045    /**
046     * Returns an expression representing the tangent of
047     * the given expression.
048     */
049    public static UnaryOpExpression tan(Expression expr) {
050        return new UnaryOpExpression(TANGENT, expr);
051    }
052
053    /**
054     * Returns an expression representing the arc sine of
055     * the given expression.
056     */
057    public static UnaryOpExpression asin(Expression expr) {
058        return new UnaryOpExpression(ARC_SINE, expr);
059    }
060
061    /**
062     * Returns an expression representing the arc cosine of
063     * the given expression.
064     */
065    public static UnaryOpExpression acos(Expression expr) {
066        return new UnaryOpExpression(ARC_COSINE, expr);
067    }
068
069    /**
070     * Returns an expression representing the arc tangent of
071     * the given expression.
072     */
073    public static UnaryOpExpression atan(Expression expr) {
074        return new UnaryOpExpression(ARC_TANGENT, expr);
075    }
076}
