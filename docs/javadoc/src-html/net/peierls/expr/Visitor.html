001package net.peierls.expr;
002
003/**
004 * Visitor interface for computing values from known expression subtypes.
005 * Traversal of composite expression is accomplished by calling
006 * {@link Expression#accept Expression.accept(Vistor<R>)}
007 * on subexpressions within visit method implementations.
008 */
009public interface Visitor<R> {
010    /** Apply this visitor function to the given constant expression. */
011    R visit(ConstantExpression expr);
012
013    /** Apply this visitor function to the given variable expression. */
014    R visit(VariableExpression expr);
015
016    /** Apply this visitor function to the given unary operation expression. */
017    R visit(UnaryOpExpression expr);
018
019    /** Apply this visitor function to the given binary operation expression. */
020    R visit(BinaryOpExpression expr);
021
022    /** Apply this visitor function to the given bound expression. */
023    R visit(LetExpression expr);
024
025    /** Apply this visitor function to the given expression of unknown type. */
026    R visitUnknown(Expression expr);
027}
