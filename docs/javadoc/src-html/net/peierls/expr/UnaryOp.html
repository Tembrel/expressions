001package net.peierls.expr;
002
003/**
004 * A double precision unary operator, supporting evaluation and
005 * formatting (string representation).
006 */
007public interface UnaryOp extends Operator {
008
009    /**
010     * Apply this operation to a double value.
011     */
012    double evaluate(double v);
013
014    /**
015     * Find the string representation of this operation given a string
016     * representation of its argument.
017     */
018    String format(String s);
019}
