001package net.peierls.expr;
002
003import java.util.Objects;
004
005
006/**
007 * An expression consisting of a constant value.
008 */
009public class ConstantExpression extends Expression {
010
011    private static final ConstantExpression PI = new ConstantExpression(Math.PI);
012    private static final ConstantExpression E = new ConstantExpression(Math.E);
013
014    private final double value;
015
016    ConstantExpression(double value) {
017        if (!Double.isFinite(value)) {
018            throw new IllegalArgumentException("constant expressions must be finite");
019        }
020        this.value = value;
021    }
022
023
024    /**
025     * Constant expression for the value of pi.
026     */
027    public static ConstantExpression pi() {
028        return PI;
029    }
030
031    /**
032     * Constant expression for the value of e.
033     */
034    public static ConstantExpression e() {
035        return E;
036    }
037
038    /** The double precision value represented by this constant expression. */
039    @Override public double value() {
040        return value;
041    }
042
043    @Override protected <R> R accept(Visitor<R> visitor) {
044        return visitor.visit(this);
045    }
046
047    @Override public boolean equals(Object obj) {
048        if (this == obj) return true;
049        if (!(obj instanceof ConstantExpression)) return false;
050        double otherValue = ((ConstantExpression) obj).value;
051        return Objects.equals(value, otherValue);
052    }
053
054    @Override public int hashCode() {
055        return Objects.hash(value);
056    }
057}
