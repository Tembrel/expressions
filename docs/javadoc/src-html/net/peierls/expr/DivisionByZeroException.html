001package net.peierls.expr;
002
003/** Thrown on a division by zero during evaluation */
004public class DivisionByZeroException extends EvaluationException {
005    final double numerator;
006    final double denominator;
007    DivisionByZeroException(double numerator, double denominator) {
008        super(String.format("Division by zero or NaN during evaluation: %f/%f", numerator, denominator));
009        this.numerator = numerator;
010        this.denominator = denominator;
011    }
012    public double numerator() { return numerator; }
013    public double denominator() { return denominator; }
014}
