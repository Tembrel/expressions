001package net.peierls.expr;
002
003import com.google.common.collect.ImmutableMap;
004
005import java.util.Map;
006import java.util.Objects;
007
008
009/**
010 * An expression representing a subexpression with variable names
011 * bound to other expressions.
012 */
013public class LetExpression extends Expression {
014    private final Expression expr;
015    private final ImmutableMap<String, Expression> bindings;
016
017    LetExpression(Expression expr, Map<String, Expression> bindings) {
018        this.bindings = ImmutableMap.copyOf(bindings);
019        this.expr = expr;
020    }
021
022    /** The subexpression bound in this expression. */
023    public Expression subExpression() {
024        return expr;
025    }
026
027    /** The local bindings of this bound expression. */
028    public Map<String, Expression> bindings() {
029        return bindings;
030    }
031
032    @Override protected <R> R accept(Visitor<R> visitor) {
033        return visitor.visit(this);
034    }
035
036    @Override public boolean equals(Object obj) {
037        if (this == obj) return true;
038        if (!(this instanceof LetExpression)) return false;
039        LetExpression that = (LetExpression) obj;
040        return this.bindings.equals(that.bindings)
041            && this.expr.equals(that.expr);
042    }
043
044    @Override public int hashCode() {
045        return Objects.hash(bindings, expr);
046    }
047}
