001package net.peierls.expr;
002
003import java.util.Objects;
004
005
006/**
007 * An expression representing the result of a binary operation on two subexpressions.
008 */
009public class BinaryOpExpression extends OperationExpression {
010    private final BinaryOp op;
011    private final Expression left;
012    private final Expression right;
013
014    BinaryOpExpression(BinaryOp op, Expression left, Expression right) {
015        this.op = op;
016        this.left = left;
017        this.right = right;
018    }
019
020    /** The binary operation used in thie expression. */
021    @Override public BinaryOp operator() {
022        return op;
023    }
024
025    /** The left argument of this expression's binary operation. */
026    public Expression leftExpression() {
027        return left;
028    }
029
030    /** The right argument of this expression's binary operation. */
031    public Expression rightExpression() {
032        return right;
033    }
034
035    @Override protected <R> R accept(Visitor<R> visitor) {
036        return visitor.visit(this);
037    }
038
039    @Override public boolean equals(Object obj) {
040        if (this == obj) return true;
041        if (!(obj instanceof BinaryOpExpression)) return false;
042        BinaryOpExpression that = (BinaryOpExpression) obj;
043        return this.op.equals(that.op)
044            && this.left.equals(that.left)
045            && this.right.equals(that.right);
046    }
047
048    @Override public int hashCode() {
049        return Objects.hash(op, left, right);
050    }
051}
