001package net.peierls.expr;
002
003import java.util.Objects;
004
005
006/**
007 * An expression representing the result of a unary operation on a subexpression.
008 */
009public class UnaryOpExpression extends OperationExpression {
010    private final UnaryOp op;
011    private final Expression expr;
012
013    UnaryOpExpression(UnaryOp op, Expression expr) {
014        this.op = op;
015        this.expr = expr;
016    }
017
018    /** The unary operation used in thie expression. */
019    @Override public UnaryOp operator() {
020        return op;
021    }
022
023    /** The argument of this expression's unary operation. */
024    public Expression subExpression() {
025        return expr;
026    }
027
028    @Override protected <R> R accept(Visitor<R> visitor) {
029        return visitor.visit(this);
030    }
031
032    @Override public boolean equals(Object obj) {
033        if (this == obj) return true;
034        if (!(obj instanceof UnaryOpExpression)) return false;
035        UnaryOpExpression that = (UnaryOpExpression) obj;
036        return this.op.equals(that.op)
037            && this.expr.equals(that.expr);
038    }
039
040    @Override public int hashCode() {
041        return Objects.hash(op, expr);
042    }
043}
