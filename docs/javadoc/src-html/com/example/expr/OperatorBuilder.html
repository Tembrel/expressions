001package com.example.expr;
002
003import static com.example.expr.Operator.*;
004
005import java.util.function.DoubleBinaryOperator;
006import java.util.function.DoubleUnaryOperator;
007
008/**
009 *
010 */
011public class OperatorBuilder implements UnaryOp, BinaryOp {
012
013    final DoubleUnaryOperator unaryOp;
014    final DoubleBinaryOperator binaryOp;
015    final String symbol;
016    final int precedence;
017    final Type type;
018
019
020    OperatorBuilder(DoubleUnaryOperator unaryOp, DoubleBinaryOperator binaryOp,
021            String symbol, int precedence, Type type) {
022        this.unaryOp = unaryOp;
023        this.binaryOp = binaryOp;
024        this.symbol = symbol;
025        this.precedence = precedence;
026        this.type = type;
027    }
028
029
030    /**
031     * Creates a builder for a unary prefix operator.
032     */
033    public static OperatorBuilder prefix(String symbol, DoubleUnaryOperator unaryOp) {
034        return unary(symbol, unaryOp, Type.PREFIX);
035    }
036
037    /**
038     * Creates a builder for a unary postfix operator.
039     */
040    public static OperatorBuilder postfix(String symbol, DoubleUnaryOperator unaryOp) {
041        return unary(symbol, unaryOp, Type.POSTFIX);
042    }
043
044    /**
045     * Creates a builder for a left-associative binary operator.
046     */
047    public static OperatorBuilder infixl(String symbol, DoubleBinaryOperator binaryOp) {
048        return binary(symbol, binaryOp, Type.INFIXL);
049    }
050
051    /**
052     * Creates a builder for a non-associative binary operator.
053     */
054    public static OperatorBuilder infixn(String symbol, DoubleBinaryOperator binaryOp) {
055        return binary(symbol, binaryOp, Type.INFIXN);
056    }
057
058    /**
059     * Creates a builder for a right-associative binary operator.
060     */
061    public static OperatorBuilder infixr(String symbol, DoubleBinaryOperator binaryOp) {
062        return binary(symbol, binaryOp, Type.INFIXR);
063    }
064
065    /**
066     * Creates a builder for a unary operator of the given type.
067     */
068    private static OperatorBuilder unary(String symbol, DoubleUnaryOperator unaryOp, Type type) {
069        if (unaryOp == null) {
070            throw new IllegalArgumentException("operator argument must not be null");
071        }
072        if (type.arity() != 1) {
073            throw new IllegalArgumentException("type argument must be PREFIX or POSTFIX");
074        }
075        // XXX ensure valid symbol
076        return new OperatorBuilder(unaryOp, null, symbol, 0, type);
077    }
078
079    /**
080     * Creates a builder for a binary operator of the given type.
081     */
082    private static OperatorBuilder binary(String symbol, DoubleBinaryOperator binaryOp, Type type) {
083        if (binaryOp == null) {
084            throw new IllegalArgumentException("operator argument must not be null");
085        }
086        if (type.arity() != 2) {
087            throw new IllegalArgumentException("type argument must be INFIXL, INFIXN, or INFIXR");
088        }
089        // XXX ensure valid symbol
090        return new OperatorBuilder(null, binaryOp, symbol, 0, type);
091    }
092
093
094    /**
095     * Sets the precedence of this operator.
096     */
097    public OperatorBuilder precedence(int precedence) {
098        return new OperatorBuilder(this.unaryOp, this.binaryOp, this.symbol, precedence, this.type);
099    }
100
101
102    @Override public String symbol() {
103        return symbol;
104    }
105
106    @Override public int precedence() {
107        return precedence;
108    }
109
110    @Override public Type type() {
111        return type;
112    }
113
114
115    @Override public double evaluate(double v) {
116        return unaryOp.applyAsDouble(v);
117    }
118
119    @Override public String format(String s) {
120        switch (type()) {
121            case PREFIX:
122                return String.format("%s%s", symbol, s);
123            case POSTFIX:
124                return String.format("%s%s", s, symbol);
125            default:
126                throw new IllegalStateException("Unary operator must be prefix or postfix");
127        }
128    }
129
130    @Override public double evaluate(double v1, double v2) {
131        return binaryOp.applyAsDouble(v1, v2);
132    }
133
134    @Override public String format(String s1, String s2) {
135        switch (type()) {
136            case INFIXL:
137            case INFIXN:
138            case INFIXR:
139                return String.format("%s%s%s", s1, symbol, s2);
140            default:
141                throw new IllegalStateException("Binary operator must be infix");
142        }
143    }
144}
