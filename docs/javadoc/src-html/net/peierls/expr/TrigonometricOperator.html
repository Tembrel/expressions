001package net.peierls.expr;
002
003import static net.peierls.expr.OperatorBuilder.*;
004
005
006/**
007 * Trigonometric operations on expressions.
008 */
009@SuppressWarnings("ImmutableEnumChecker")
010public enum TrigonometricOperator implements DelegatingOp {
011
012    SINE        (prefix("sin ", Math::sin)      .precedence(100)),
013    COSINE      (prefix("cos ", Math::cos)      .precedence(100)),
014    TANGENT     (prefix("tan ", Math::tan)      .precedence(100)),
015    ARC_SINE    (prefix("asin ", Math::asin)    .precedence(100)),
016    ARC_COSINE  (prefix("acos ", Math::acos)    .precedence(100)),
017    ARC_TANGENT (prefix("atan ", Math::atan)    .precedence(100)),
018
019    ;
020
021    Operator delegate;
022
023    TrigonometricOperator(Operator delegate) {
024        this.delegate = delegate;
025    }
026
027    @Override public Operator delegate() { return delegate; }
028
029
030    public static UnaryOpExpression sin(Expression expr) {
031        return new UnaryOpExpression(SINE, expr);
032    }
033
034    public static UnaryOpExpression cos(Expression expr) {
035        return new UnaryOpExpression(COSINE, expr);
036    }
037
038    public static UnaryOpExpression tan(Expression expr) {
039        return new UnaryOpExpression(TANGENT, expr);
040    }
041
042    public static UnaryOpExpression asin(Expression expr) {
043        return new UnaryOpExpression(ARC_SINE, expr);
044    }
045
046    public static UnaryOpExpression acos(Expression expr) {
047        return new UnaryOpExpression(ARC_COSINE, expr);
048    }
049
050    public static UnaryOpExpression atan(Expression expr) {
051        return new UnaryOpExpression(ARC_TANGENT, expr);
052    }
053}
